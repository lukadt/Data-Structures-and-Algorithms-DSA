/* Euclid's algorithm to compute greatest common denominator */
algorithm Gcd(m, n)
	Pre:	m and n are integers
	Post:	the greatest common denominator of the two integers is calculated
	if n = 0 
		return m // the value of m is the gcd.
	end if
	return Gcd(n, m) // a recursive call to the Gcd method swapping the parameters around
end Gcd
